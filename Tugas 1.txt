1. Masuk ke MariaDB : 
- MySQL -u root -p

2. Membuat database baru (dbpos):
- CREATE database dbpos;

3. Gunakan database dbpos : 
- USE dbpos;

4. Membuat Tabel baru (Kartu):
- CREATE TABLE kartu (
- id integer auto_increment primary key,
- kode varchar(4) unique,
- nama varchar(30) not null,
- diskon double default 0,
- iuran double default 0 );

5. Membuat Tabel baru (Pelanggan):
- CREATE TABLE pelanggan (
- id int auto_increment primary key,
- kode varchar(10) unique,
- nama varchar(45),
- jk char(1),
- tmp_lahir varchar(30),
- tgl_lahir date,
- email varchar(45),
- kartu_id int not null references kartu(id) );

6. Membuat Tabel baru (Pesanan):
- CREATE TABLE pesanan (
- id int auto_increment primary key,
- tanggal date,
- total double,
- pelanggan_id int not null references pelanggan(id) );

7. Membuat Tabel baru (Pembayaran) :
- CREATE TABLE pembayaran (
- id int not null auto_increment primary key,
- tanggal date,
- jumlah double,
- ke int,
- pesanan_id int not null references pesanan(id) );

8. Membuat Tabel baru (Jenis Produk) :
- CREATE TABLE jenis_produk (
- id int not null auto_increment primary key,
- nama varchar(45) );

9. Membuat Tabel baru (Produk) :
- CREATE TABLLE produk (
- id int not null auto_increment primary key,
- kode varchar(10) unique,
- nama varchar(45),
- harga_beli double,
- harga_jual double,
- stok int,
- min_stok int,
- jenis_produk int not null references jenis_produk(id) );

10. Membuat Tabel baru (Pesanan Items) :
- CREATE TABLE pesanan_items (
- id int not null auto_increment primary key,
- produk_id int not null references produk(id),
- pesanan_id int not null references pesanan(id),
- qty int,
- harga double );

11. Membuat Tabel baru (Vendor) :
- CREATE TABLE vendor (
- id int not null auto_increment primary key,
- nomor varchar(4) unique,
- nama varchar(40),
- kota varchar(30),
- kontak varchar(30) );

12. Membuat Tabel baru (Pembelian) :
- CREATE TABLE pembeliam (
- id int not null auto_increment primary key,
- tanggal varchar(45),
- nomor varchar(10) unique,
- produk_id int not null references produk(id),
- jumlah int,
- harga double,
- vendor_id int not null references vendor(id) );
